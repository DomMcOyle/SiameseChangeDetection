 164: def contrastive_loss(y_true, y_pred):
 165:     """
 166:     Function implementing the contrastive loss for the training phase
 167:     :param y_true: the actual value for the pixel's class (1 not changed = same class, 0 changed = different class)
 168:     :param y_pred: the predicted value for the pixel's class
 169:     :return: the value of the contrastive loss. The margin, positive value which helps to make largely
 170:     dissimilar pairs to count toward the loss computation, is stored in config.py
 171:     """
 172:     y_true = tf.cast(y_true, y_pred.dtype)
 173:     square_pred = kb.square(y_pred)
 174:     square_margin = kb.square(kb.maximum(config.MARGIN - y_pred, 0))
 175:     return kb.mean(y_true * square_pred + (1 - y_true) * square_margin)
 176: 
 177: def SAM(tens):
 178:     """
 179:     Function defining the Spectral Angle Mapper score to be applied
 180:     on the result of the Siamese Branches
 181:     :param tens: the two tensors (spectral signature of each pixel) to be compared
 182:     :return: the SAM score between the tensors as arccos(t1xt2/norm2(t1)*norm2(t2))
 183:     """
 184:     (x, y) = tens
 185:     xnorm = kb.l2_normalize(x, axis=1)
 186:     ynorm = kb.l2_normalize(y, axis=1)
 187:     dot = kb.sum(xnorm * ynorm, axis=1, keepdims=True)
 188:     # dot must be bounded since some values could a little bit more than 1 or less than -1
 189:     dot = kb.maximum(dot, -1)
 190:     dot = kb.minimum(dot, 1)
 191:     return tf.math.acos(dot)

 193: def accuracy(y_true, y_pred):
 194:     """
 195:     Function computing the training accuracy
 196:     :param y_true: the actual value for the pixel's class (1 not changed = same class, 0 changed = different class)
 197:     :param y_pred: the predicted value for the pixel's class
 198:     :return: the accuracy considering y_pred = 1 <=> y_pred<0.5
 199:     """
 200:     return kb.mean(kb.equal(y_true, kb.cast(y_pred < config.PRED_THRESHOLD, y_true.dtype)))
